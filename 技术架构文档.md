## 3. 数据库设计

### 3.1 表结构

```sql
-- 对话表
CREATE TABLE conversations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    parent_id UUID REFERENCES conversations(id),
    name VARCHAR(255) NOT NULL,
    status VARCHAR(50) DEFAULT 'active',  -- active/exploring/done/archived
    fork_from UUID REFERENCES conversations(id),
    fork_text TEXT,
    auto_exploring BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 消息表
CREATE TABLE messages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations(id),
    role VARCHAR(50) NOT NULL,  -- user/assistant/system
    content TEXT NOT NULL,
    node_type VARCHAR(50) DEFAULT 'normal',  -- normal/annotation/merge
    created_at TIMESTAMP DEFAULT NOW()
);

-- 消息关系表（支持 DAG）—— 唯一的关系存储方式
-- 决策：采用独立关系表而非 messages 表内 parents 字段
-- 原因：外键约束保证完整性，支持按 relation_type 过滤，删除时自动级联
CREATE TABLE message_relations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    parent_id UUID NOT NULL REFERENCES messages(id),
    child_id UUID NOT NULL REFERENCES messages(id),
    relation_type VARCHAR(50) DEFAULT 'normal',  -- normal/fork/merge
    created_at TIMESTAMP DEFAULT NOW()
);

-- 标注表
CREATE TABLE annotations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    message_id UUID NOT NULL REFERENCES messages(id),
    text VARCHAR(255) NOT NULL,
    suggestions JSONB DEFAULT '[]',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 合并记录表
CREATE TABLE merge_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    target_id UUID NOT NULL REFERENCES conversations(id),
    source_ids UUID[] NOT NULL,
    conclusion TEXT NOT NULL,
    keep_option VARCHAR(50) DEFAULT 'keep',
    created_at TIMESTAMP DEFAULT NOW()
);
```

## 5. 核心算法

### 5.1 分叉建议算法
```python
def should_suggest_fork(message_content: str, annotations: List) -> bool:
    """
    判断是否建议分叉
    - 标注数量 >= 3
    - 或检测到对比/选择类关键词
    - 或信息密度超过阈值
    """
    if len(annotations) >= 3:
        return True

    keywords = ['对比', '选择', '哪个', '还是', '或者', 'vs', 'versus']
    if any(kw in message_content for kw in keywords):
        return True

    return False
```

### 5.2 自主探索算法
```python
async def auto_explore_branch(branch_id: str, max_depth: int):
    """
    在分支内自主探索
    - 设定探索目标
    - AI 自问自答推进
    - 检测收敛条件
    - 达到深度或目标达成后停止
    """
    for round in range(max_depth):
        # 生成问题
        question = await generate_question(branch_id)

        # 生成回答
        answer = await llm.chat(question, context=branch_id)

        # 保存消息
        save_message(branch_id, 'assistant', answer)

        # 检测是否收敛
        if await is_converged(branch_id):
            break

    mark_branch_done(branch_id)
```

### 5.3 合并综合算法
```python
def merge_conclusions(target_id: str, source_ids: List[str]) -> str:
    """
    综合多个分支结论
    - 提取各分支核心结论
    - 去重、排序
    - 生成综合摘要
    """
    conclusions = []
    for sid in source_ids:
        conv = get_conversation(sid)
        conclusion = extract_conclusion(conv)
        conclusions.append(conclusion)

    # 调用 LLM 综合
    merged = llm.synthesize(conclusions)
    return merged
```
